// 데이터 불러오기 로직 수정 function loadData() { const params = new URLSearchParams(window.location.search); const target = params.get('target'); if (target) { document.getElementById('targetNumber').value = target; initialPrefix = target.slice(0, 2); // 타겟값 기반 접두사 업데이트 } // 모든 베드 데이터 로드 Array.from(document.querySelectorAll('.input-numbers')).forEach((input, index) => { const bedData = params.get(`bed${index+1}`); if (bedData) { // 쉼표(,)로 구분된 데이터를 개행문자(\n)로 변환 input.value = bedData.split(',').join('\n'); calculateResults(input); // 결과 재계산 } else { input.value = ''; // 데이터 없을 경우 초기화 calculateResults(input); } }); updateAverage(); // 평균 업데이트 }
